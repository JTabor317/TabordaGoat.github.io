Unity / C# Intro to Game Dev Final Project

Windowed

A puzzle game based on the icons in the corner of a program window. 2D environment with a character that has the ability to push, close, resize, and minimize blocks to solve the puzzles.

I worked on designing and implementing the core puzzle systems, focusing on how players interact with the environment through both character movement and direct cursor input. The main challenge was creating intuitive mechanics based on familiar UI symbols (minimize, close, resize) while ensuring they translated into meaningful and solvable puzzle interactions. To address this, I helped develop a grid-based system that simplified movement and interactions, reducing edge cases and allowing for more controlled puzzle design. I also contributed to implementing core mechanics such as clickable blocks, state-changing buttons, and environmental constraints like glass barriers, all of which were introduced progressively to guide player understanding. One key design consideration was preventing soft-lock scenarios, which led to the addition of a reset mechanic and more intentional level design. This project was interesting to work on as it had the challenge of conveying what UI elements would do in a gameplay scenerio while still making them feel satisfying and fun to interact with.

C# movement snippet

A short sample is enough to show style and clarity.

public void TryStep(Vector2 inputDir)
{
if (isMoving || mytilemap == null)
    retu
// snap to cardinal direction
Vector2Int dir = ToCardinal(inputDir);
if (dir == Vector2Int.zero)
    retu
// current cell
Vector3Int currentCell = mytilemap.WorldToCell(transform.position);
Vector3Int nextCell = currentCell + new Vector3Int(dir.x, dir.y, 0); 
// wall check
if (IsBlocked(nextCell))
    retu
// check for pushable
Pushable blockInFront = FindPushableInCell(nextCell);

if (blockInFront != null)
{
    Clickable cl = blockInFront.GetComponent(); //get a clickable behavior bc all the stuff is stored in that i guess
    // if not pushable, cant move
    if ((cl == null) || (!cl.IsMinimized()))
    {
        Vector2Int nextBlockScalers = new Vector2Int(1,1);
        if (cl != null) //Remove Old Effect Tile Behavior
        {
            ButtonEffectTile p = FindEffectTileInCell(nextCell);
            if (p != null)
                cl.setOnEffectTile(p.getActiveType(),false);
            nextBlockScalers = new Vector2Int((int)cl.getCurrentScalers().x, (int)cl.getCurrentScalers().y); //i hope this works
        }  
        
        Vector3Int blockMoveTargetCall = new Vector3Int((int)math.floor(blockInFront.transform.position.x), (int)math.floor(blockInFront.transform.position.y) , 0) + new Vector3Int(dir.x, dir.y, 0);
        Vector3Int blockWallTargetCell = nextCell + new Vector3Int(dir.x * nextBlockScalers.x, dir.y * nextBlockScalers.y, 0);
        if ((IsBlocked(blockWallTargetCell) || FindPushableInCell(blockWallTargetCell) != null))
            retu
        // move block if pushable
        Vector3 blockTargetPos = mytilemap.GetCellCenterWorld(blockMoveTargetCall);
        StartCoroutine(MoveObject(blockInFront.transform, blockTargetPos, false));
        if (cl != null) //Set New Effect Tile Behavior
        {
            ButtonEffectTile p = FindEffectTileInCell(blockMoveTargetCall);
            if (p != null)
                cl.setOnEffectTile(p.getActiveType(),true);
        }   
    }
}

public static IEnumerable<string> ValidateEncounter(EncounterData data)
{
    if (data.SpawnPoints.Count == 0)
    {
        yield return "Encounter needs at least one spawn point.";
    }

    if (data.Waves.Any(wave => wave.Enemies.Count == 0))
    {
        yield return "Each wave needs at least one enemy entry.";
    }
}
